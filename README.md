### 2. 找出不是两个数组中的共同元素
```c#include <stdio.h>
int main(){
    int m,n;
    int i,j;
    scanf("%d",&m);
    int a[m];
    for(i=0;i<m;i++){
        scanf("%d",&a[i]);
    }
    scanf("%d",&n);
    int b[n];
    for(i=0;i<n;i++){
        scanf("%d",&b[i]);
    }
    int c[m];
    c[0]=a[0];
    int k=0,flag=0;
    for(i=1;i<m;i++){
        flag=0;
        for(j=0;j<=k;j++){
            if(a[i]==c[j]){
                flag=1;
            }
        }
        if(flag==0){
            k++;
            c[k]=a[i];
        }
    }
    int d[n];
    d[0]=b[0];
    int z=0;
        for(i=1;i<n;i++){
        flag=0;
        for(j=0;j<=z;j++){
            if(b[i]==d[j]){
                flag=1;
            }
        }
        if(flag==0){
            z++;
            d[z]=b[i];
        }
    }
    flag=0;
   for(i=0;i<=k;i++){
   	flag=0;
       for(j=0;j<=z;j++){
           if(c[i]==d[j]){
               flag=1;
           }
       }
      if(flag==0){
           printf("%d ",c[i]);
       }
   }
    
   for(i=0;i<=z;i++){
   	flag=0;

}
```
